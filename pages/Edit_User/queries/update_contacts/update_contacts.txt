mutation Update($personId: String!, $data: people_contact_v0_set_input) {
  update_people_contact_v0(where: {
    _and: [
      {PrincipalFestCloudID: {_eq: $personId}}
      {EmailType: {_eq: "Personal"}}
      {PhoneType: {_eq: "Personal"}}
    ]
  }, _set: $data) {
    affected_rows
		returning{
      PhoneNumber
      Email
      EmailType
      PhoneType
    }
  }
}
